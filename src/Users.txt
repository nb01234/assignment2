private int calculatePartScore(double hrs, double a, double b) {
       
        double percent = a * b / hrs;
        int score = 0;
       
        // score out of 5 based on how much the user took breaks
        if (percent < 0.02) {
            score = 0;
        } else if (percent < 0.04) {
            score = 1;
        } else if (percent < 0.06) {
            score = 2;
        } else if (percent < 0.08) {
            score = 3;
        } else if (percent < 0.1) {
            score = 4;
        } else {
            score = 5;
        }
       
        return score;
    }
